# SearchRequest

## Example Usage

```typescript
import { SearchRequest } from "@r9s/sdk/models";

let value: SearchRequest = {
  engine: "tavily",
  extra: {
    query: "latest AI news",
    additionalProperties: {
      "max_results": 5,
      "include_answer": true,
    },
  },
};
```
